    莉･荳九・縲￣ID蛻ｶ蠕｡繧剃ｽｿ逕ｨ縺励※繝｢繝ｼ繧ｿ繝ｼ隗貞ｺｦ縺ｮ蛻ｶ蠕｡繧定｡後≧邁｡蜊倥↑萓九〒縺吶らｩ榊・縺ｮ證ｴ襍ｰ繧帝亟縺舌◆繧√↓Pid_setGainWithLimit繧剃ｽｿ逕ｨ縺励∝ｿ・ｦ√↓蠢懊§縺ｦPid_setInvert縺ｧ蜃ｺ蜉帙・隨ｦ蜿ｷ繧貞渚霆｢縺ｧ縺阪∪縺吶・/p> 

    #include "pid.h"
#include "encoder.h"
#include "motor_driver.h"
#include <stdio.h>

TIM_HandleTypeDef htim3;
TIM_HandleTypeDef htim5;
MotorDriver motor;
Encoder encoder;
EncoderData encoder_data;
Pid pid;

void main(void) {
    HAL_Init();
    SystemClock_Config();
    MX_TIM3_Init();
    MX_TIM5_Init();

    MotorDriver_Init(&motor, &htim3, TIM_CHANNEL_1, TIM_CHANNEL_2);
    Encoder_Init(&encoder, &htim5, 100.0, 8192, 10);

    Pid_Init(&pid);
    // 豈比ｾ九・遨榊・繝ｻ蠕ｮ蛻・ご繧､繝ｳ縲∵凾螳壽焚縲∫ｩ榊・隱､蟾ｮ縺ｮ荳企剞繧定ｨｭ螳夲ｼ井ｾ・ ﾂｱ100・・/span>
    Pid_setGainWithLimit(&pid, 1.0, 0.1, 0.01, 0.0, 100.0);
    // 蠢・ｦ√↓蠢懊§縺ｦ蜃ｺ蜉帙・隨ｦ蜿ｷ繧貞渚霆｢・・: 豁｣蟶ｸ, -1: 蜿崎ｻ｢・・/span>
    Pid_setInvert(&pid, 1);

    double target_angle = 90.0;  // 逶ｮ讓呵ｧ貞ｺｦ (蠎ｦ)

    while (1) {
        Encoder_Interrupt(&encoder, &encoder_data);
        double current_angle = encoder_data.deg;

        double control_value = Pid_control(&pid, target_angle, current_angle, 10);

        MotorDriver_setSpeed(&motor, (int)control_value);

        // 繝・ヰ繝・げ蜃ｺ蜉・/span>
        printf("Target Angle: %.2f, Current Angle: %.2f, Control Value: %.2f\n",
               target_angle, current_angle, control_value);

        HAL_Delay(10);  // 蛻ｶ蠕｡蜻ｨ譛・10ms
    }
}
